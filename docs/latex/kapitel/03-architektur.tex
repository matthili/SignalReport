\subsection{Key-Design-Entscheidungen}
\begin{itemize}
    \item \textbf{Kein separates Backend}: Alle Komponenten in einer Anwendung (einfachere Installation)
    \item \textbf{H2-Datenbank}: Embedded-DB für plattformunabhängige Installation
    \item \textbf{Javalin statt Spring Boot}: Minimaler Overhead für Web-Interface
    \item \textbf{Java 21}: Nutzung moderner Features wie Virtual Threads
\end{itemize}
